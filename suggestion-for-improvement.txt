async def langgraph_chatbot(user_message: str, user_id: str = None, session_id: str = None):
    """Handle user message with LangGraph memory"""
    try:
        logger.info(f"User message from user {user_id}: {user_message}")

        # Use thread_id for checkpointing (combine user_id + session_id)
        # For separate contexts per tab:
        thread_id = f"user_{user_id}_session_{session_id}"
        # For shared context across tabs, use: thread_id = f"user_{user_id}"

        # Config for checkpointer
        config = {
            "configurable": {
                "thread_id": thread_id
            }
        }

        # Check if we have existing state
        existing_state = None
        try:
            # Try to get existing checkpoint
            existing_state = await graph.checkpointer.aget(config)
        except Exception as e:
            logger.info(f"No existing checkpoint found for thread {thread_id}: {e}")

        # Initial state
        if existing_state and "state" in existing_state:
            # Use existing state and add new message
            state = existing_state["state"]
            state["messages"].append(HumanMessage(content=user_message))
        else:
            # Create new state
            state = {
                "user_id": user_id,
                "session_id": session_id,
                "messages": [HumanMessage(content=user_message)],
                "route": "router_node"
            }

        # Stream with checkpointing
        async def run_langgraph_with_memory():
            new_ai_messages = []

            async for event in graph.astream(state, config=config):
                for value in event.values():
                    # Extract new AI messages
                    if "messages" in value:
                        for msg in value["messages"]:
                            if isinstance(msg, AIMessage) and msg.content:
                                logger.info(f"Assistant: {msg.content}")
                                new_ai_messages.append(msg.content)
                            elif isinstance(msg, ToolMessage) and msg.content:
                                logger.info(f"[Tool Result] {msg.content}")
                                new_ai_messages.append(msg.content)

            return new_ai_messages

        new_ai_messages = await run_langgraph_with_memory()

        # Return response
        return "\n".join(new_ai_messages) if new_ai_messages else "No response."

    except Exception as e:
        logger.error(f"Error in langgraph_chatbot: {e}")
        raise ExceptionError(e)

# Optional: Session management utilities
async def clear_user_session(user_id: str, session_id: str):
    """Clear user session from checkpoints"""
    try:
        thread_id = f"user_{user_id}_session_{session_id}"
        config = {
            "configurable": {
                "thread_id": thread_id
            }
        }
        await graph.checkpointer.adelete(config)
        logger.info(f"Cleared session for user {user_id}, session {session_id}")
    except Exception as e:
        logger.error(f"Error clearing session: {e}")
